Дополнительные особенности:
- желательно не пользоваться интернетом для поиска решений во время проведения соревнований;
- при подаче решения не забывайте выбрать корректный язык программирования;
- после нажатия на клавишу "Отправить" не нужно ждать окончания проверки поданного решения - переходите к следующей задаче, проверка длится несколько секунд;
- за каждую неудачную попытку накладывается штраф, равный 20 минутам;
- в каждой задаче от 10 до 30 тестов, если ваша программа хотя бы на один тест даст неправильный ответ, то на экране задачи будет надпись WA (WrongAnswer) с указанием номера теста;
- если вы выбрали некорректный язык программирования или допустили синтаксическую ошибку, то получите надпись CE (ошибка компилятора) или RE (ошибка времени выполнения);
- монитор положения участников будет доступен постоянно во время проведения соревнования;
- кто решит больше задач, тот займёт лучшее место, но, при равенстве количества решённых задач, сравнивается накопленное время с учётом штрафов, например, если вы решили две задачи - одну на 10 минуте, вторую на 15 минуте но со второй попытки, то итоговое время будет такое: 10 + 15 + 20 = 45. У кого накопленное время меньше, тот займёт лучшее место.
Примеры оформления программы поиска суммы чисел на разных языках: 
// C# - на входе в одной строке два числа

    using System;
    using System.Linq;

    // namespace можно не указывать
    class Program
    {
        static void Main() // аргументы можно не указывать
        {
            Console.WriteLine(
                Console
                    .ReadLine()
                    .Split()
                    .Select(x=>int.Parse(x))
                    .Sum()
            );
            Console.ReadLine(); // эту строку можно оставить
        }
    }
                
# Python - на входе в одной строке два числа

    print(sum(map(int, input().split())))
        
# Java - на входе в одной строке два числа

    import java.util.Scanner;

    class Main {
        public static void main(String[] args) {
            Scanner sc = new Scanner(System.in);
            String line = sc.nextLine();
            String[] nums = line.split(" ");
            int result = Integer.parseInt(nums[0]) + Integer.parseInt(nums[1]);
	        System.out.println(result);
        } 
    }
        
# C++ - на входе два числа

    #include 
    using namespace std;

    int aplusb(int x, int y) {
        return x + y;
    }

    int main() {
        int a, b;
        cin >> a >> b;
        cout << aplusb(a, b);
    }
        
// Node.js - на вход подаются две строки, в каждой одно число

    var readline = require('readline');

    var rl = readline.createInterface({
        input: process.stdin,
        output: process.stdout
    });
    
    let lines = [];
    
    rl.on('line', function(line){
        lines.push(line);
    });
    
    rl.on('close', () => {
        result = +lines[0] + +lines[1];
        console.log(result);
    });
